Data Cleaning

some functions I made to make a new variable. Not perfect, but good enough.

neg_binom <- function(bi_var) {
  counts <- numeric(length(bi_var))
  current_count <- 0
  for (i in seq_along(bi_var)) {
    if (bi_var[i] == 0) {
      current_count <- current_count + 1  
      counts[i] <- current_count
    } else {
      current_count <- 0
    }
  }
  
  return(counts)
}


impute <- function(df, var) {
  previous_value <- NA
  for (i in 1:nrow(df)) {
    if (is.na(df[i, var])) {
      df[i, var] <- previous_value
    } else {
      previous_value <- df[i, var]
    }
  }
  return(df)
}

Random Forest

Okay, so the MERF function was not working, but the MERT (tree) model was, so I had to improvise to get some result. I first wrote code that would randomly sample 3 of the variable in the data set to run a tree on. I then selected n_group number of observations from each group with replacement, with n_group being the number of observation in the orginal dataset. I then fit the random forest, and saved the random effects in a matrix and repeated this 500 times.

REmat <- matrix(NA, nrow = 196, ncol = 500)

for (i in 1:500) {
  
  variable_to_select <- dat %>%
  select(-COWcode, -year, -v2x_polyarchy, -countryn) %>%
  colnames()
  
  variables <- sample(variable_to_select, 3, replace = FALSE)
  
  sampled_data <- dat %>%
    group_by(COWcode) %>%
    sample_n(size = n(), replace = TRUE) %>%
    arrange(COWcode, year) %>%
    mutate(n_in_group = row_number()) %>%
    ungroup()
  
  X <- sampled_data %>%
    select(all_of(variables))
  
  list_dat <- list(
    X = as.matrix(X),
    Y = as.numeric(pull(sampled_data, v2x_polyarchy)),
    id = as.numeric(pull(sampled_data, COWcode)),
    Z = as.matrix(rep(1, nrow(sampled_data))),
    time = as.numeric(pull(sampled_data, n_in_group)))


  mert1 <- LongituRF::MERT(
    X = data.frame(list_dat$X),
    Y = list_dat$Y,
    id = list_dat$id,
    Z = list_dat$Z,
    time = list_dat$time,
    sto = 'OrnUhl')

REmat[, i] <- mert1$random_effects

}

Here I clean the data to be presented in a Caterpillar plot.

REmat <- as.matrix(REmat)

REmat <- t(REmat)

REmat <- as.data.frame(REmat)

data_viz <- REmat %>%
  summarize_all(list(
    lower = ~quantile(., probs = 0.025, na.rm = TRUE),
    mean = ~mean(., na.rm = TRUE),
    upper = ~quantile(., probs = 0.975, na.rm = TRUE)
  ))

data_viz_long <- data_viz %>%
  pivot_longer(cols = everything(), 
               names_to = c("variable", ".value"),
               names_sep = "_") %>%
  mutate(problem = ifelse(mean > upper | mean < lower, FALSE, TRUE),
         mean = as.numeric(mean),
         lower = as.numeric(lower),
         upper = as.numeric(upper),
         sig = ifelse((lower > 0 & upper > 0 | lower < 0 & upper < 0), T, F))
